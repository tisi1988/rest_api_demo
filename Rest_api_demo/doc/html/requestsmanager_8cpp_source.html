132 QByteArray RequestsManager::decryptPayload(const qint64 contentId, QAESEncryption::Mode aes_mode, QAESEncryption::Aes key_length){
133  Content content_data;
134  QString error_message;
135  bool res = db_controller_->getContent(contentId, &content_data, &error_message);
136  if(!res){
137  throw CustomException(QString("Could not get content data from the database: %1").arg(error_message));

139  QAESEncryption encryption(key_length, aes_mode);

141  QByteArray decodedText = encryption.decode(QByteArray::fromHex(content_data.getPayloadData()), key);
142  decodedText = encryption.removePadding(decodedText);

158 bool RequestsManager::validateProtectionSystem(const qint64 &deviceId, const qint64 &contentId){
159  // Validate if the Content Protection System is the same as the device protection system. If it is
160  // to proceed to the next step, if not, return an error.
161  Content content_data;
